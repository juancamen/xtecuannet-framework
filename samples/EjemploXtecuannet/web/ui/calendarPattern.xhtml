Calendar - Date Pattern

By default date pattern is MM/dd/yyy. Custom patterns can be applied using the pattern attribute.

dd.MM.yyyy

yy, M, d

EEE, dd MMM, yyyy

Source

<h:form>
			
	<h3>dd.MM.yyyy</h3>
	<p:calendar value="\#{calendarBean.date1}" pattern="dd.MM.yyyy" />	
	
	<h3>yy, M, d</h3>
	<p:calendar value="\#{calendarBean.date2}" pattern="yy, M, d"/>
	
	<h3>EEE, dd MMM, yyyy</h3>
	<p:calendar value="\#{calendarBean.date3}" pattern="EEE, dd MMM, yyyy"/>

    <p:commandButton value="Submit" update="display" oncomplete="dialog.show()"/>
	
	<p:dialog header="Selected Dates" widgetVar="dialog" 
				showEffect="fold" hideEffect="fold"
				height="150">
	
		<h:panelGrid id="display" columns="2" cellpadding="5">
			<h:outputText value="dd.MM.yyyy: " />

				<f:convertDateTime pattern="dd.MM.yyyy" />

package org.primefaces.examples.view;

import java.util.Date;

public class CalendarBean {

	private Date date1;
	
	private Date date2;
	
	private Date date3;
	
	public Date getDate1() {
		return date1;
	}

	public void setDate1(Date date1) {
		this.date1 = date1;
	}

	public Date getDate2() {
		return date2;
	}

	public void setDate2(Date date2) {
		this.date2 = date2;
	}
	
	public Date getDate3() {
		return date3;
	}

	public void setDate3(Date date3) {
		this.date3 = date3;
	}
}
